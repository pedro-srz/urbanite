47 {
48  if (display_id == PORT_REAR_PARKING_DISPLAY_ID)
49  {
50  /* TIMER PWM: TIM4 and CHANNEL1 (red), CHANNEL2 (green), CHANNEL3 (blue) (we must change 5 registers) */
51  /* TO-DO alumnos */
52  RCC->APB1ENR |= RCC_APB1ENR_TIM4EN; // 1) First! Enable the peripheral clock
53  TIM4->CR1 &= ~TIM_CR1_CEN; // 2) Disable counter (optional).
54  TIM4->CR1 = TIM_CR1_ARPE; // Autoreload preload enabled
55  TIM4->CNT = 0; // 3) Reset counter
56  TIM4->ARR = 63999; // 4) Load the autoreload value
57  TIM4->PSC = 4; // 5) Load the prescaler
58 
59  /*
60  PWM (hay que tocar 5 registros)
61  */
62  // PWM --> miramos en el pinout o la "Table 11. Alternate function"
63  // Deshabilitar el bit TIM_CCER_CC1E del registro CCER antes de hacer modificaciones (no es indispensable).
64  // * Deshabilita el Output Compare --> CCER &= ~TIM_CCER_CC1E
65  // Output compare 1 --> tenemos que mirar registro TIMx_CCMR1
66  // * Por defecto es salida, si no, deberíamos poner --> CCxS = 0x0
67  // * Modo PWM1 --> OCxM = 6 // 0b0110
68  // * Habilitamos el preload --> OCxPE = 1
69  // Cargamos el valor del ancho del pulso en los bits CCR1 del registro TIMx_CCR1 (canales 1 y 2) o TIMx_CCR2 (canales 3 y 4)
70  // * DC (%) = CCRx/ARR
71  // --> donde CCRx es CCR1 (canal 1), y ARR es el valor del registro TIMx_ARR
72  // --> para un ciclo de trabajo del 50% --> CCR1 = (ARR + 1) * 0.5
73 
74  // Rojo: TIM4_CH1, Verde: TIM4_CH3, Azul: TIM4_CH4
75  // 1) Disable Output Compare for channel 1, 3 and 4
76  TIM4->CCER &= ~TIM_CCER_CC1E; // Channel 1
77  TIM4->CCER &= ~TIM_CCER_CC3E; // Channel 3
78  TIM4->CCER &= ~TIM_CCER_CC4E; // Channel 4
79 
80  // 1) Set non inverted PWM mode 1 for channel 1, 3 and 4 (P and NP bits are cleared)
81  TIM4->CCER &= ~(TIM_CCER_CC1P | TIM_CCER_CC1NP); // Channel 1
82  TIM4->CCER &= ~(TIM_CCER_CC3P | TIM_CCER_CC3NP); // Channel 3
83  TIM4->CCER &= ~(TIM_CCER_CC4P | TIM_CCER_CC4NP); // Channel 4
84 
85  // It is also possible to use the following line to clear the bits: TIM4->CCER &= ~(TIM_CCER_CC1E | TIM_CCER_CC1P | TIM_CCER_CC1NP); or TIM4->CCER = 0; // Clear all bits of the register considering that Channel 2 is not used and does not care

126 void port_display_set_rgb(uint32_t display_id, rgb_color_t color){
127  // Extract the color values
128  uint8_t r = color.r;
129  uint8_t g = color.g;
130  uint8_t b = color.b;
131 
132  // El CCRx adapta el ciclo de trabajo. Esta parte del duty cycle la rehacemos cada vez que se cambia de displaya aquí
133  if (display_id == PORT_REAR_PARKING_DISPLAY_ID)
134  {
135  // Disable the timer
136  TIM4->CR1 &= ~TIM_CR1_CEN;
137  TIM4->CNT = 0;
138 
139  if (r == 0 && g == 0 && b == 0)
140  {
141  TIM4->CCER &= ~TIM_CCER_CC1E; // Disable the output compare for channel 1
142  TIM4->CCER &= ~TIM_CCER_CC3E; // Disable the output compare for channel 3
143  TIM4->CCER &= ~TIM_CCER_CC4E; // Disable the output compare for channel 4
144  return;
145  }
146 
147  if (r == 0)
148  {
149  TIM4->CCER &= ~TIM_CCER_CC1E; // Disable the output compare for channel 1
150  }
151  else
152  {
153  // Habilitamos el output compare en el registro CCER --> CCxE = 1
154  // 4) PWM pulse width
155  TIM4->CCR1 = (uint32_t)((((TIM4->ARR + 1) * r) / PORT_DISPLAY_RGB_MAX_VALUE) - 1); // Red led (channel 1)
156  TIM4->CCER |= TIM_CCER_CC1E; // Enable the output compare for channel 1
157  }
158 
159  if (g == 0)
160  {
161  TIM4->CCER &= ~TIM_CCER_CC3E; // Disable the output compare for channel 3
162  }
163  else
164  {
165  TIM4->CCR3 = (uint32_t)((((TIM4->ARR + 1) * g) / PORT_DISPLAY_RGB_MAX_VALUE) - 1); // Green led (channel 3)
166  TIM4->CCER |= TIM_CCER_CC3E; // Enable the output compare for channel 3
167  }
168 
169  if (b == 0)
170  {
171  TIM4->CCER &= ~TIM_CCER_CC4E; // Disable the output compare for channel 4
172  }
173  else
174  {
175  TIM4->CCR4 = (uint32_t)((((TIM4->ARR + 1) * b) / PORT_DISPLAY_RGB_MAX_VALUE) - 1); // Blue led (channel 4)
176  TIM4->CCER |= TIM_CCER_CC4E; // Enable the output compare for channel 4
177  }
178 
179  // The PSC and ARR values are currently in the preload registers. To load them into the active registers we
180  // need an update event. We can do this manually as follows (or we could wait for the timer to expire).
181  TIM4->EGR |= TIM_EGR_UG; // 6) Update generation: Re-inicializa el contador y actualiza los registros. IMPORTANTE que esté lo último
182 
183  // Enable the timer
184  TIM4->CR1 |= TIM_CR1_CEN;
185  }
