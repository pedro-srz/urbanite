144 static void do_set_distance(fsm_t *p_this)
145 {
146  fsm_ultrasound_t *p_fsm = (fsm_ultrasound_t *)(p_this);
147 
148  uint32_t echo_init_tick = port_ultrasound_get_echo_init_tick(p_fsm->ultrasound_id);
149  uint32_t echo_end_tick = port_ultrasound_get_echo_end_tick(p_fsm->ultrasound_id);
150  uint32_t echo_overflows = port_ultrasound_get_echo_overflows(p_fsm->ultrasound_id);
151 
152  uint32_t time_go_back_us = (echo_end_tick - echo_init_tick) + (echo_overflows * 65536);
153 
154  // Calculate distance in cm taking into account the speed of sound (1cm = 58us)
155  uint32_t distance = (time_go_back_us * SPEED_OF_SOUND_MS * 100) / (2 * 1000000); // 2 because the sound goes back and forth
156 
157  // Store the distance in the array
158  p_fsm->distance_arr[p_fsm->distance_idx] = distance;
159 
160  if (p_fsm->distance_idx >= FSM_ULTRASOUND_NUM_MEASUREMENTS - 1)
161  {
162  // If the array is full, compute the distance
163  // Compute the median
164  // Sort the array
165  qsort(p_fsm->distance_arr, FSM_ULTRASOUND_NUM_MEASUREMENTS, sizeof(uint32_t), _compare);
166 
167  if (FSM_ULTRASOUND_NUM_MEASUREMENTS % 2 == 0)
168  {
169  p_fsm->distance_cm = (p_fsm->distance_arr[FSM_ULTRASOUND_NUM_MEASUREMENTS / 2 - 1] +
170  p_fsm->distance_arr[FSM_ULTRASOUND_NUM_MEASUREMENTS / 2]) /
171  2;
172  }
173  else
174  {
175  p_fsm->distance_cm = p_fsm->distance_arr[FSM_ULTRASOUND_NUM_MEASUREMENTS / 2];
176  }
177 
178  p_fsm->new_measurement = true; // Flag to indicate the system that a new measurement is ready
179  }
180 
181  // Increase distance modulo the number of MAX_MEASUREMENTS
182  p_fsm->distance_idx = (p_fsm->distance_idx + 1) % FSM_ULTRASOUND_NUM_MEASUREMENTS;
183
