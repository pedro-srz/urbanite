54 static void system_clock_config(void)
55 {
57  /* Power controller (PWR) */
58  /* Control the main internal voltage regulator output voltage to achieve a trade-off between performance and power consumption when the device does not operate at the maximum frequency */
59  PWR->CR &= ~PWR_CR_VOS; // Clean and set value
60  PWR->CR |= (PWR_CR_VOS & (POWER_REGULATOR_VOLTAGE_SCALE3 << PWR_CR_VOS_Pos));
61 
62  /* Initializes the RCC Oscillators. */
63  /* Adjusts the Internal High Speed oscillator (HSI) calibration value.*/
64  RCC->CR &= ~RCC_CR_HSITRIM; // Clean and set value
65  RCC->CR |= (RCC_CR_HSITRIM & (RCC_HSI_CALIBRATION_DEFAULT << RCC_CR_HSITRIM_Pos));
66 
67  /* RCC Clock Config */
68  /* Initializes the CPU, AHB and APB buses clocks */
69  /* To correctly read data from FLASH memory, the number of wait states (LATENCY)
70  must be correctly programmed according to the frequency of the CPU clock
71  (HCLK) and the supply voltage of the device. */
72 
73  /* Increasing the number of wait states because of higher CPU frequency */
74  FLASH->ACR = FLASH_ACR_LATENCY_2WS; /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
75 
76  /* Change in clock source is performed in 16 clock cycles after writing to CFGR */
77  RCC->CFGR &= ~RCC_CFGR_SW; // Clean and set value
78  RCC->CFGR |= (RCC_CFGR_SW & (RCC_CFGR_SW_HSI << RCC_CFGR_SW_Pos));
79 
80  /* Update the SystemCoreClock global variable */
81  SystemCoreClock = HSI_VALUE >> AHBPrescTable[(RCC->CFGR & RCC_CFGR_HPRE) >> RCC_CFGR_HPRE_Pos];
82 
83  /* Configure the source of time base considering new system clocks settings */
84  SysTick_Config(SystemCoreClock / (1000U / TICK_FREQ_1KHZ)); /* Set Systick to 1 ms */
85 }

120 uint32_t port_system_init()
121 {
122 
123 #ifdef USE_SEMIHOSTING
124  initialise_monitor_handles();
125 #endif
126 
127  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
128  /* Configure Flash prefetch, Instruction cache, Data cache */
129  /* Instruction cache enable */
130  FLASH->ACR |= FLASH_ACR_ICEN;
131 
132  /* Data cache enable */
133  FLASH->ACR |= FLASH_ACR_DCEN;
134 
135  /* Prefetch cache enable */
136  FLASH->ACR |= FLASH_ACR_PRFTEN;
137 
138  /* Set Interrupt Group Priority */
139  NVIC_SetPriorityGrouping(NVIC_PRIORITY_GROUP_4);
140 
141  /* Use systick as time base source and configure 1ms tick (default clock after Reset is HSI) */
142  /* Configure the SysTick IRQ priority. It must be the highest (lower number: 0)*/
143  NVIC_SetPriority(SysTick_IRQn, NVIC_EncodePriority(NVIC_GetPriorityGrouping(), 0U, 0U)); /* Tick interrupt priority */
144 
145  /* Init the low level hardware */
146  /* Reset and clock control (RCC) */
147  RCC->APB2ENR |= RCC_APB2ENR_SYSCFGEN; /* Syscfg clock enabling */
148 
149  /* Peripheral clock enable register */
150  RCC->APB1ENR |= RCC_APB1ENR_PWREN; /* PWREN: Power interface clock enable */
151 
152  /* Configure the system clock */
153  system_clock_config();
154 
155  return 0;
156 }

196 void stm32f4_system_gpio_config(GPIO_TypeDef *p_port, uint8_t pin, uint8_t mode, uint8_t pupd)
197 {
198  if (p_port == GPIOA)
199  {
200  RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN; /* GPIOA_CLK_ENABLE */
201  }
202  else if (p_port == GPIOB)

204  RCC->AHB1ENR |= RCC_AHB1ENR_GPIOBEN; /* GPIOB_CLK_ENABLE */
205  }
206  else if (p_port == GPIOC)
207  {
208  RCC->AHB1ENR |= RCC_AHB1ENR_GPIOCEN; /* GPIOC_CLK_ENABLE */
209  }
210 
211  /* Clean ( &=~ ) by displacing the base register and set the configuration ( |= ) */
212  p_port->MODER &= ~(GPIO_MODER_MODER0 << (pin * 2U));
213  p_port->MODER |= (mode << (pin * 2U));
214 
215  p_port->PUPDR &= ~(GPIO_PUPDR_PUPD0 << (pin * 2U));
216  p_port->PUPDR |= (pupd << (pin * 2U));

219 void stm32f4_system_gpio_config_exti(GPIO_TypeDef *p_port, uint8_t pin, uint32_t mode)
220 {
221  uint32_t port_selector = 0;
222 
223  RCC->APB2ENR |= RCC_APB2ENR_SYSCFGEN;
224 
225  /* SYSCFG external interrupt configuration register */
226  if (p_port == GPIOA)
227  {
228  port_selector = 0;
229  }
230  else if (p_port == GPIOB)
231  {
232  port_selector = 1;
233  }
234  else if (p_port == GPIOC)
235  {
236  port_selector = 2;
237  }
238 
239  uint32_t base_mask = 0x0FU;
240  uint32_t displacement = (pin % 4) * 4;
241 
242  SYSCFG->EXTICR[pin / 4] &= ~(base_mask << displacement);
243  SYSCFG->EXTICR[pin / 4] |= (port_selector << displacement);
244 
245  /* Rising trigger selection register (EXTI_RTSR) */
246  EXTI->RTSR &= ~BIT_POS_TO_MASK(pin);
247  if (mode & STM32F4_TRIGGER_RISING_EDGE)
248  {
249  EXTI->RTSR |= BIT_POS_TO_MASK(pin);
250  }
251 
252  /* Falling trigger selection register (EXTI_FTSR) */
253  EXTI->FTSR &= ~BIT_POS_TO_MASK(pin);
254  if (mode & STM32F4_TRIGGER_FALLING_EDGE)
255  {
256  EXTI->FTSR |= BIT_POS_TO_MASK(pin);
257  }
258 
259  /* Event mask register (EXTI_EMR) */
260  EXTI->EMR &= ~BIT_POS_TO_MASK(pin);
261  if (mode & STM32F4_TRIGGER_ENABLE_EVENT_REQ)
262  {
263  EXTI->EMR |= BIT_POS_TO_MASK(pin);
264  }
265 
266  /* Clear EXTI line configuration */
267  EXTI->IMR &= ~BIT_POS_TO_MASK(pin);
268 
269  /* Interrupt mask register (EXTI_IMR) */
270  if (mode & STM32F4_TRIGGER_ENABLE_INTERR_REQ)
271  {
272  EXTI->IMR |= BIT_POS_TO_MASK(pin);
273  }
274 }

335 void port_system_power_stop()
336 {
337  /* Select the regulator state in Stop mode: Set PDDS and LPDS bits according to PWR_Regulator value */
338  MODIFY_REG(PWR->CR, (PWR_CR_PDDS | PWR_CR_LPDS), PWR_CR_LPDS);
339 
340  /* Set SLEEPDEEP bit of Cortex System Control Register */
341  SCB->SCR |= ((uint32_t)SCB_SCR_SLEEPDEEP_Msk);
342 
343  /* Select Stop mode entry : Request Wait For Interrupt */
344  __WFI();
345 
346  /* Reset SLEEPDEEP bit of Cortex System Control Register */
347  SCB->SCR &= ~((uint32_t)SCB_SCR_SLEEPDEEP_Msk);
