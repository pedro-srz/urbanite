55 static void _timer_trigger_setup(uint32_t ultrasound_id)
56 {
57  if (ultrasound_id == PORT_REAR_PARKING_SENSOR_ID)
58  {
59  RCC->APB1ENR |= RCC_APB1ENR_TIM3EN; // 1) Hay que habilitar la fuente de reloj. IMPORTANTE que esté lo primero
60 
61  // Por defecto la fuente es interna (bits CKD=00 in TIM3->CR1 register)
62  TIM3->CR1 &= ~TIM_CR1_CEN; // 2) Disable counter. (no es indispensable).
63  TIM3->CR1 |= TIM_CR1_ARPE; // Autoreload preload enabled
64 
65  TIM3->SR &= ~TIM_SR_UIF; // Limpiamos interrupciones pendientes
66  TIM3->CNT = 0;
67 
68  // Set the trigger pin to high for a short period of time depending on the duration in microseconds
69  // Compute ARR and PSC to match the duration in microseconds. Check if the duration is too long and adapt prescaler and ARR
70  double scc = (double)SystemCoreClock;
71  double psc = round((((scc / 1000000.0) * PORT_REAR_PARKING_SENSOR_TRIGGER_UP_US) / (65535.0 + 1.0)) - 1.0);
72  double arr = round((((scc / 1000000.0) * PORT_REAR_PARKING_SENSOR_TRIGGER_UP_US) / (psc + 1.0)) - 1.0);
73 
74  // Adjust psc and arr if necessary
75  if (arr > 0xFFFF)
76  {
77  psc += 1.0;
78  arr = round(((scc / 1000000.0) * PORT_REAR_PARKING_SENSOR_TRIGGER_UP_US) / (psc + 1.0) - 1.0);
79  }
80 
81  // Load the values
82  TIM3->ARR = (uint32_t)(round(arr));
83  TIM3->PSC = (uint32_t)(round(psc));
84 
85  TIM3->DIER |= TIM_DIER_UIE; // Habilita las interrupciones en el registro
86

99 static void _timer_echo_setup(uint32_t ultrasound_id)
100 {
101  if (ultrasound_id == PORT_REAR_PARKING_SENSOR_ID)
102  {
103  RCC->APB1ENR |= RCC_APB1ENR_TIM2EN; // Enable the clock of timer 4
104 
105  // Set TIM2 Channel 2 as input capture
106  TIM2->PSC = 15; // Set up an appropriate prescaler to slow down the clock or timer counter. 1 us
107  TIM2->ARR = 0xFFFF; // Set auto-reload register to 65499
108 
109  // Set the direction as input and select the active input
110  // CC2S[1:0] for channel 2;
111  // 00 = output
112  // 01 = input, CC1 is mapped on timer Input 1
113  // 10 = input, CC1 is mapped on timer Input 2
114  // 11 = input, CC1 is mapped on slave timer
115  TIM2->CCMR1 &= ~TIM_CCMR1_CC2S;
116  TIM2->CCMR1 |= (0x1 << TIM_CCMR1_CC2S_Pos);
117 
118  // Disable digital filtering by clearing IC2F[3:0] bits
119  // because we want to capture every event
120  TIM2->CCMR1 &= ~TIM_CCMR1_IC2F;
121 
122  // Select the edge of the active transition
123  // Detect only rising edges in this example
124  // CC2NP:CC2P bits
125  // 00 = rising edge,
126  // 01 = falling edge,
127  // 10 = reserved,
128  // 11 = both edges
129  TIM2->CCER |= (1 << TIM_CCER_CC2P_Pos | 1 << TIM_CCER_CC2NP_Pos); // Both rising and falling edges.
130 
131  // Program the input prescaler
132  // To capture each valid transition, set the input prescaler to zero;
133  // IC2PSC[1:0] bits (input capture 1 prescaler)
134  TIM2->CCMR1 &= ~(TIM_CCMR1_IC2PSC); // Clear filtering because we need to capture every event
135 
136  // Enable Capture/compare output enable for channel 2
137  TIM2->CCER |= TIM_CCER_CC2E;
138 
139  // Enable related interrupts
140  TIM2->DIER |= TIM_DIER_CC2IE; // Enable Capture/Compare interrupts for channel 2
141  TIM2->DIER |= TIM_DIER_UIE; // Enable update interrupts
142 
143  // DO NOT ENABLE THE COUNTER YET
144 
145  NVIC_SetPriority(TIM2_IRQn, 1); // Set priority to 1
146  NVIC_EnableIRQ(TIM2_IRQn); // Enable TIM2 interrupt in NVIC
147  }

164  double ms = (double)PORT_REAR_PARKING_SENSOR_TIMEOUT_MS;

168  double arr = round((((scc / 1000.0) * ms) / (psc + 1.0)) - 1.0);
169

172  {
173  psc += 1.0;
